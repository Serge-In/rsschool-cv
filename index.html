                '2021-04-02 построить шкалу времени

                 Function diagScaleTime(aCandles, CandleStep, ScaleTimeStepTime)

                 

                    'массив подписей шкалы времени для массива aCandles

                     ReDim aScaleTime(UBound(aCandles, 1), 2)

                  

                     'ожидаемое время первой подписи

                      CandleTimeNext = aCandles(1, 2) 
                    

                        For i = 1 To UBound(aCandles, 1)

                    

                            If aCandles(i, 2) >= CandleTimeNext Then

                                'создать подпись (время)
    
                                 aScaleTime(i, 1) = aCandles(i, 2) 
 
                                 
 
                                 '+ мин шаг между подписями

                                 i = i + CandleStep - 1 
 
                                 
 
                                 'ожидаемое время следующей подписи
 
                                 CandleTimeNext = CandleTimeNext + ScaleTimeStepTime 

                            End If

                        Next i

                    

                    diagScaleTime = aScaleTime 'вернуть массив подписей

                End Function
